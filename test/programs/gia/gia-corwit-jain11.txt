OBSERVER AUTOMATON AssumptionAutomaton

INITIAL STATE N14;

STATE __TRUE :
    TRUE -> GOTO __TRUE;

STATE __qTEMP :
    TRUE -> ASSUME {false} GOTO __qTEMP;

STATE __qFINAL :
    TRUE -> GOTO __qFINAL;

STATE USEALL N24 :
    MATCH "__VERIFIER_assert (y!=0U);" -> GOTO N5;
    MATCH OTHERWISE -> GOTO N24;
    TRUE -> GOTO __qFINAL;
    INVARIANT {(( y % 2 ) == 1 )};

STATE USEALL N5 :
    MATCH "[!(!(cond))]" -> GOTO N10;
    MATCH "[!(cond)]" -> GOTO N8;

    TRUE -> GOTO __qFINAL;

    MATCH OTHERWISE -> GOTO N10;
    TRUE -> GOTO __qFINAL;

    MATCH OTHERWISE -> GOTO N4;
    TRUE -> GOTO __qFINAL;

STATE USEALL N8 :

    MATCH OTHERWISE -> GOTO N8;
    TRUE -> GOTO __qFINAL;

STATE USEALL N1 :
    MATCH "" -> GOTO N10;
    MATCH OTHERWISE -> GOTO N1;
    TRUE -> GOTO __qFINAL;

STATE USEALL N14 :

    MATCH OTHERWISE -> GOTO N14;
    TRUE -> GOTO __qFINAL;

STATE USEALL N15 :

    MATCH OTHERWISE -> GOTO N15;
